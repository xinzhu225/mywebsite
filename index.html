from selenium import webdriver from selenium.webdriver.chrome.service import Service from selenium.webdriver.chrome.options import Options from bs4 import BeautifulSoup from urllib.parse import urljoin, urlparse import time import os import csv import re # ---------- 配置 ---------- chrome_driver_path = "/path/to/chromedriver" # 替换为你的 chromedriver 路径 start_url = "https://你的网址.com" # 替换成你自己的网址 output_folder = "full_site_content" csv_file = "website_summary_paragraphs.csv" wait_time = 5 # 每页等待 JS 渲染的秒数 # ---------- 创建输出文件夹 ---------- os.makedirs(output_folder, exist_ok=True) # ---------- 设置 Selenium ---------- chrome_options = Options() chrome_options.add_argument("--headless") # 无头模式 chrome_options.add_argument("--disable-gpu") chrome_options.add_argument("--no-sandbox") chrome_options.add_argument("--window-size=1920,1080") service = Service(chrome_driver_path) driver = webdriver.Chrome(service=service, options=chrome_options) # ---------- 初始化抓取队列 ---------- to_crawl = [start_url] crawled = set() domain = urlparse(start_url).netloc # ---------- CSV 文件初始化 ---------- csv_path = os.path.join(output_folder, csv_file) with open(csv_path, "w", encoding="utf-8", newline="") as f: writer = csv.writer(f) writer.writerow(["URL", "标题", "段落序号", "段落内容", "链接列表"]) # ---------- 开始抓取 ---------- while to_crawl: url = to_crawl.pop(0) if url in crawled: continue try: driver.get(url) time.sleep(wait_time) html = driver.page_source soup = BeautifulSoup(html, 'html.parser') crawled.add(url) print(f"正在抓取: {url} （已抓取 {len(crawled)} 个页面）") # ---------- 页面内容 ---------- title = soup.title.string.strip() if soup.title else "无标题" paragraphs = [p.get_text(strip=True) for p in soup.find_all('p') if p.get_text(strip=True)] links = [] for link in soup.find_all('a', href=True): href = link['href'] text = link.get_text(strip=True) full_url = urljoin(url, href) links.append(f"{text} -> {full_url}") # 只抓取同域名内部链接 if urlparse(full_url).netloc == domain and full_url not in crawled and full_url not in to_crawl: to_crawl.append(full_url) links_text = "\n".join(links) # ---------- 保存单页面文件 ---------- # 文件夹名安全处理 safe_name = re.sub(r"[^\w\-_.]", "_", urlparse(url).path.strip("/")) or "root" page_folder = os.path.join(output_folder, safe_name) os.makedirs(page_folder, exist_ok=True) with open(os.path.join(page_folder, "title.txt"), "w", encoding="utf-8") as f: f.write(title) with open(os.path.join(page_folder, "paragraphs.txt"), "w", encoding="utf-8") as f: f.write("\n\n".join(paragraphs)) with open(os.path.join(page_folder, "links.txt"), "w", encoding="utf-8") as f: f.write(links_text) # ---------- 写入 CSV，每段文字一行 ---------- with open(csv_path, "a", encoding="utf-8", newline="") as f: writer = csv.writer(f) for i, paragraph in enumerate(paragraphs, start=1): writer.writerow([url, title, i, paragraph, links_text]) except Exception as e: print(f"抓取失败: {url}，原因: {e}") driver.quit() print(f"抓取完成，总共抓取 {len(crawled)} 个页面，CSV 文件已生成: {csv_file}")
