import express from "express"; import fetch from "node-fetch"; const app = express(); app.use(express.json()); app.post("/v1/chat/completions", async (req, res) => { try { const response = await fetch( "https://api.openai.com/v1/chat/completions", { method: "POST", headers: { "Content-Type": "application/json", Authorization: "Bearer YOUR_OPENAI_KEY", }, body: JSON.stringify(req.body), } ); const data = await response.json(); res.json(data); } catch (err) { res.status(500).json({ error: err.message }); } }); app.listen(3000, () => { console.log("API 中转站已启动"); });
